// P1
$<< n1;
$<< n2;
$<< n3;

var max: number = n1;

fi (n2 > max) {
    max = n2;
}

fi (n3 > max) {
    max = n3;
}

$>> max;

// P2
$<< n;

fi (n == 2) {
    $>> `Prime`;
}

fi (n mod 2 == 0) {
    $>> `Not prime`;
}

var sqaureRoot: number = rad(n);

let i in seq(3, sqaureRoot, 2) {
    fi (n mod i == 0) {
        $>> `Not prime`;
    }
}

$>> `Prime`;


// P3

var array: vector<number>;
var len: number;

$<< len;

let i in seq(len) {
    $<< array[i];
}

var sum: number = 0;

loop (len) {
    dec len;
    sum += $array[len];
}   

$>> sum;
